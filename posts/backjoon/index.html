n = int(input())

max_l = 0
max_num = []

for i in range(1, n+1):
    numbers = [n, i]

    while numbers[-1] >= 0:
        numbers.append(numbers[-2] - numbers[-1])

    numbers.pop()
    
    if max_l < len(numbers):
        max_l = len(numbers)
        max_num = numbers

print(max_l)
print(*max_num)
